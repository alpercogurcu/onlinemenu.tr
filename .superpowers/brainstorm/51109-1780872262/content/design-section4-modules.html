ProviderConfig — tüm modüllerde ortak temel tip
type ProviderConfig struct {
  Provider string
  TenantID uuid.UUID
  BranchID *uuid.UUID // nil=tenant
  Config map[string]any
  VaultSecretPath string
  Environment string // test|prod
  IsActive bool
}
AdapterFactory[T] — generic factory
type AdapterFactory[T any] func(
  cfg ProviderConfig,
  creds map[string]string, // Vault'tan
) (T, error)

// Kullanım (delivery modülünde):
factory.Register("yemeksepeti",
  yemeksepeti.NewAdapter)
factory.Register("getir",
  getir.NewAdapter)

DeliveryAdapter interface (public/)
type DeliveryAdapter interface {

  // Inbound
  ParseWebhook(
    body []byte,
    headers map[string]string,
  ) (WebhookEvent, error)

  ConfirmOrder(ctx, externalID string) error
  RejectOrder(ctx, externalID, reason string) error

  // Outbound
  UpdateAvailability(
    ctx,
    products []ProductAvailability,
  ) error

  SyncMenu(ctx, snapshot CatalogSnapshot) error
  UpdateHours(ctx, hours BranchHours) error
}

Event Sözleşmeleri — delivery modülü
delivery.order.received.v1 → pos modülü tüketir → orders tablosuna yazar, kasa WebSocket bildirim
delivery.order.accepted.v1 → adapter.ConfirmOrder çağrılır, platform'a kabul gönderilir
delivery.order.rejected.v1 → adapter.RejectOrder, platforma ret gönderilir
catalog.product_availability.changed.v1 → delivery subscriber tüketir → aktif tüm platformlara broadcast (UpdateAvailability)
catalog.product.updated.v1 → delivery subscriber → SyncMenu (fiyat/isim değişikliği)
tenant.branch_hours.updated.v1 → delivery subscriber → UpdateHours tüm aktif platformlara
